# Categorizing and visualizing blood pressure levels
def bp_category(systolic):
    if systolic < 120:
        return 'Normal'
    elif systolic < 130:
        return 'Elevated'
    elif systolic < 140:
        return 'Hypertension Stage 1'
    else:
        return 'Hypertension Stage 2'
data['bp_category'] = data['systolic_bp'].apply(bp_category)
bp_counts = data['bp_category'].value_counts()
bp_counts.plot(kind='pie', autopct='%1.1f%%')
plt.title('Blood Pressure Categories')
plt.ylabel('')
plt.show()

This pie chart categorizes individuals into different blood pressure groups, helping provide us a clear visual of the population's blood pressure distribution.
